109inline u8 gsuInit(void) {
110 gsu_cfgr = 0x80; /* IRQ mask, no fast multiply */
111 gsu_scmr = 0x19; /* 4bpp + RAN + RON (most common for PLOT) */
112 gsu_scbr = 0x00; /* Buffer A */
113 gsu_dma_src_hi = 0x00; /* DMA from buffer A */
114 return superfx_status != 0;
115}
