635/*============================================================================
636 * Fast Macro Sprite API
637 *
638 * Zero-overhead alternatives to oamSet/oamSetXY for performance-critical code.
639 * These write directly to oamMemory[] without function call overhead.
640 *
641 * oamSet() has framesize=158 per call due to SSA temporaries. With >2-3
642 * sprites/frame in the main loop, the stack manipulation causes visible
643 * jitter. These macros eliminate that overhead entirely.
644 *
645 * Note: cc65816 does not truly inline 'static inline' functions — they
646 * become separate SUPERFREE sections with global labels that conflict
647 * across translation units. Macros are the only zero-overhead option.
648 *
649 * oamMemory[] and oam_update_flag are declared in <snes/system.h>
650 * (included automatically via <snes.h>).
651 *
652 * Usage:
653 * // Pre-compute attribute byte once at init
654 * u8 attr = OAM_ATTR(tile, palette, priority, flags);
655 *
656 * // Per-frame: fast full update
657 * oamSetFast(id, x, y, tile, palette, priority, flags);
658 *
659 * // Per-frame: position-only update (most common)
660 * oamSetXYFast(id, x, y);
661 *============================================================================*/
662
672#define OAM_ATTR(_tile, _pal, _prio, _fl) \
673 ((u8)(((_fl) & 0xC0) | \
674 (((_prio) & 0x03) << 4) | \
675 (((_pal) & 0x07) << 1) | \
676 (((_tile) >> 8) & 0x01)))
677
684#define OAM_XHI_MASK(_slot) \
685 ((u8)((_slot) == 0 ? 0x01 : (_slot) == 1 ? 0x04 : \
686 (_slot) == 2 ? 0x10 : 0x40))
687
703#define oamSetFast(_id, _x, _y, _tile, _pal, _prio, _fl) do { \
704 u16 _off = (u16)(_id) << 2; \
705 oamMemory[_off + 0] = (u8)((_x) & 0xFF); \
706 oamMemory[_off + 1] = (u8)((_y) & 0xFF); \
707 oamMemory[_off + 2] = (u8)((_tile) & 0xFF); \
708 oamMemory[_off + 3] = OAM_ATTR(_tile, _pal, _prio, _fl); \
709 u16 _ext = 512 + ((u16)(_id) >> 2); \
710 u16 _sl = (u16)(_id) & 0x03; \
711 u8 _xhi = OAM_XHI_MASK(_sl); \
712 if ((_x) & 0x100) \
713 oamMemory[_ext] |= _xhi; \
714 else \
715 oamMemory[_ext] &= ~_xhi; \
716 oam_update_flag = 1; \
717} while(0)
718
729#define oamSetXYFast(_id, _x, _y) do { \
730 u16 _off = (u16)(_id) << 2; \
731 oamMemory[_off + 0] = (u8)((_x) & 0xFF); \
732 oamMemory[_off + 1] = (u8)((_y) & 0xFF); \
733 u16 _ext = 512 + ((u16)(_id) >> 2); \
734 u16 _sl = (u16)(_id) & 0x03; \
735 u8 _xhi = OAM_XHI_MASK(_sl); \
736 if ((_x) & 0x100) \
737 oamMemory[_ext] |= _xhi; \
738 else \
739 oamMemory[_ext] &= ~_xhi; \
