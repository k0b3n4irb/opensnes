697/*============================================================================
698 * Fast Macro Sprite API
699 *
700 * Zero-overhead alternatives to oamSet/oamSetXY for performance-critical code.
701 * These write directly to oamMemory[] without function call overhead.
702 *
703 * oamSet() has framesize=158 per call due to SSA temporaries. With >2-3
704 * sprites/frame in the main loop, the stack manipulation causes visible
705 * jitter. These macros eliminate that overhead entirely.
706 *
707 * Note: cc65816 does not truly inline 'static inline' functions — they
708 * become separate SUPERFREE sections with global labels that conflict
709 * across translation units. Macros are the only zero-overhead option.
710 *
711 * oamMemory[] and oam_update_flag are declared in <snes/system.h>
712 * (included automatically via <snes.h>).
713 *
714 * Usage:
715 * // Pre-compute attribute byte once at init
716 * u8 attr = OAM_ATTR(tile, palette, priority, flags);
717 *
718 * // Per-frame: fast full update
719 * oamSetFast(id, x, y, tile, palette, priority, flags);
720 *
721 * // Per-frame: position-only update (most common)
722 * oamSetXYFast(id, x, y);
723 *============================================================================*/
724
734#define OAM_ATTR(_tile, _pal, _prio, _fl) \
735 ((u8)(((_fl) & 0xC0) | \
736 (((_prio) & 0x03) << 4) | \
737 (((_pal) & 0x07) << 1) | \
738 (((_tile) >> 8) & 0x01)))

746#define OAM_XHI_MASK(_slot) \
747 ((u8)((_slot) == 0 ? 0x01 : (_slot) == 1 ? 0x04 : \
748 (_slot) == 2 ? 0x10 : 0x40))

765#define oamSetFast(_id, _x, _y, _tile, _pal, _prio, _fl) do { \
766 u16 _off = (u16)(_id) << 2; \
767 oamMemory[_off + 0] = (u8)((_x) & 0xFF); \
768 oamMemory[_off + 1] = (u8)(((_y) - 1) & 0xFF); /* compensate +1 PPU scanline quirk */ \
769 oamMemory[_off + 2] = (u8)((_tile) & 0xFF); \
770 oamMemory[_off + 3] = OAM_ATTR(_tile, _pal, _prio, _fl); \
771 u16 _ext = 512 + ((u16)(_id) >> 2); \
772 u16 _sl = (u16)(_id) & 0x03; \
773 u8 _xhi = OAM_XHI_MASK(_sl); \
774 if ((_x) & 0x100) \
775 oamMemory[_ext] |= _xhi; \
776 else \
777 oamMemory[_ext] &= ~_xhi; \
778 oam_update_flag = 1; \
779} while(0)
780
791#define oamSetXYFast(_id, _x, _y) do { \
792 u16 _off = (u16)(_id) << 2; \
793 oamMemory[_off + 0] = (u8)((_x) & 0xFF); \
794 oamMemory[_off + 1] = (u8)(((_y) - 1) & 0xFF); /* compensate +1 PPU scanline quirk */ \
795 u16 _ext = 512 + ((u16)(_id) >> 2); \
796 u16 _sl = (u16)(_id) & 0x03; \
